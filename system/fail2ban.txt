#fail2ban
#功能：限制非法IP登陆
#参考https://idc.wanyunshuju.com/li/8.html

yum install -y shorewall gamin-python shorewall-shell shorewall-perl shorewall-common python-inotify python-ctypes fail2ban

[root@localhost fail2ban]# cat jail.local 
[DEFAULT]
bantime = 60000
findtime = 10
maxretry = 3
[sshd]
enabled = true
filter = sshd
action = iptables[name=SSH, port=ssh, protocol=tcp]
logpath = /var/log/secure
maxretry = 2

systemctl start fail2ban
systemctl enable fail2ban
